Trait d’extension pour créer des ranges numériques

Required Methods§

Source

fn to(self, end: Self) -> Builder<Self>

Crée un range de self vers end

§Exemples

// Range croissant for i in 0.to(5) { println!(“{}”, i); // 0, 1, 2, 3, 4 }

// Range décroissant for i in 5.to(0) { println!(“{}”, i); // 5, 4, 3, 2, 1 }

// Avec pas personnalisé for i in 0.to(10).by(2) { println!(“{}”, i); // 0, 2, 4, 6, 8 }
